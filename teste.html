async function analisar() { if (!foto.files.length) { alert("Escolha uma imagem."); return; } const reader = new FileReader(); reader.onload = async () => { try { const resposta = await fetch("/api/analisar", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ imagem: reader.result }) }); console.log("Status:", resposta.status); const texto = await resposta.text(); console.log("Resposta:", texto); const json = JSON.parse(texto); document.getElementById("resultado").textContent = JSON.stringify(json, null, 2); if (json.produtos && json.produtos.length) { document.getElementById("produto").style.display = "block"; let html = "

Produtos Encontrados

"; json.produtos.forEach(produto => { html += `

${produto.nome}

Marca: ${produto.marca}

Score: ${produto.score}

${ produto.url ? ` ` : "Sem URL cadastrada" }
`; }); document.getElementById("produto").innerHTML = html; } } catch (erro) { console.error(erro); document.getElementById("resultado").textContent = erro.stack || erro.message; } }; reader.readAsDataURL(foto.files[0]); }
